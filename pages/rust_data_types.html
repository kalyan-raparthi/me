COMPOND_DATATYPES

        array : fixed collection of same similar data types
        [T: N]
        
            let mut x: [u8: 3] = [1, 2, 3];
            println!("{}", a[0]); // 1
            
            tuple : fixed collection of same various data types
            (T: VALUE)
            
                let x: (u8, char, f32) = (9, 'h', 10.9); 
                println!("{}", x.0); // 9
    
                slice : a reference to contiguous sequence of same datatype elements
                &[i .. i + n]  
                
                    let a: [u8; 3] = [1, 2, 3];
    let t: &[u8] = &a[0 .. 1];
    
    str : raw array of characters
        [char: N]
    
    let c: [char: 3] = ['a', 'b', 'c'];
    let a: char = c[0];
