flow of control


    1. if, else else if : conditional code block execution
        
        let n: u8 = 1;
    
        if 1 < n {
            println!("1 < {0}", n);
        } else if 1 > n {
            println!("{0} > 1", n);
        } else {
            println!("something else");
        }
        
    2. match : swicth like control
        match VAR {
            VAR_1 => {//},
            _ => {},
        }
        
            let n: u8 = 11;
            match n {
                1 => {
                    println!("1");
                },
                2 | 3 => {
                    println!("2 | 3");
                }
                5..=10 => {
                    println!("between 5 to 10");
                }
                _ => {
                    println!("more then 10");
                }
            }
